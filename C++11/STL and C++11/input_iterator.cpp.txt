#include <iostream>
#include <iterator>
#include <algorithm>
#include <map>
#include <cctype>

using namespace std;

bool isValidChar(char c)
{
        return !(isalpha(c));
}

int main()
{
        istream_iterator<string> end;
        istream_iterator<string> ins(cin);

        map<string,int> wf;

        while(ins != end)
        {
                string word = *ins++;
                word.erase(remove_if(word.begin(),word.end(),&isValidChar),word.end());


                if(wf.find(word) != wf.end())
                {
                        wf[word]++;
                }
                else
                {
                        wf[word]=1;
                }


        }


        for(const auto& pair : wf)
        {
                cout<<pair.first<<" "<<pair.second<<"\n";
        }
        return 0;
}
