string toLowerCase(const string& s)
{
    string temp;
    std::transform(begin(s),end(s),back_insert_iterator<string>(temp),[](unsigned char c)
    {
        return tolower(c);
    });
    return temp;
}

void printData(const T& dat)
{
    for(const auto& a: dat)
    {
        cout<<a<<" ";
    }
}
