template<typename ... Ts>
auto sum(Ts... ts)
{
    return (ts + ...);
}

template <typename ... Ts>
auto product(Ts ... ts)
{
    return (ts * ... * 1);
}

int main()
{
    int the_sum {sum(1, 2, 3, 4, 5)}; // Value: 15
